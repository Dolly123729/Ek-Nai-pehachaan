// Simple chatbot logic function sendMessage() { const userInput = document.getElementById("chat-input").value.trim(); const chatBox = document.getElementById("chat-box"); if (!userInput) return; // Show user's message chatBox.innerHTML += `
${userInput}
`; document.getElementById("chat-input").value = ""; // Simple responses let botReply = ""; if (userInput.includes("hi") || userInput.includes("hello")) { botReply = "Hello! How can I help you today?"; } else if (userInput.includes("quiz")) { botReply = "You can take quizzes on the Quiz page!"; } else if (userInput.includes("ai")) { botReply = "AI tools help you study smarter. Would you like suggestions?"; } else { botReply = "Thank you for your message. I am still learning!"; } // Show bot reply setTimeout(() => { chatBox.innerHTML += `
${botReply}
`; chatBox.scrollTop = chatBox.scrollHeight; }, 500); }
